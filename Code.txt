// Analogous to: def translations(text, inputLanguage, targetLanguage)
const createTranslationPrompt = (jsonChunk, targetLanguage) => {
    return `
      As a professional translator,
      translate the values of the following JSON object into ${targetLanguage}.
      Do not translate the keys.
      Ensure medical terms are translated accurately.
      Return ONLY the valid raw JSON string without markdown formatting.

      JSON to Translate:
      ${JSON.stringify(jsonChunk)}
    `;
};

// Analogous to: def generate_output(prompt)
const generateOutput = async (prompt) => {
    try {
        const response = await fetch(
            `https://generativelanguage.googleapis.com/v1beta/models/${MODEL_ID}:generateContent?key=${apiKey}`,
            {
                method: 'POST',
                headers: { 'Content-Type': 'application/json' },
                body: JSON.stringify({
                    contents: [{ parts: [{ text: prompt }] }]
                })
            }
        );
        
        const data = await response.json();
        
        if (!data.candidates || !data.candidates[0]) {
             throw new Error("No candidates returned");
        }
        
        return data.candidates[0].content.parts[0].text;
    } catch (ex) {
        console.error("Error in generateOutput:", ex);
        throw ex;
    }
};

const LanguageContext = createContext();

const LanguageProvider = ({ children }) => {
    const [language, setLanguage] = useState('en');
    const [dir, setDir] = useState('ltr');
    const [translations, setTranslations] = useState({ en: baseTranslations });
    const [isTranslating, setIsTranslating] = useState(false);

    const fetchTranslation = async (targetLangCode, targetLangName) => {
        if (translations[targetLangCode]) return true;

        setIsTranslating(true);
        try {
            // 2. SEPARATE INTO TWO CHUNKS (Applying the concept from your request)
            const keys = Object.keys(baseTranslations);
            const midpoint = Math.ceil(keys.length / 2);
            
            // Chunk 1: First half of the keys
            const chunk1Keys = keys.slice(0, midpoint);
            const chunk1 = chunk1Keys.reduce((obj, key) => {
                obj[key] = baseTranslations[key];
                return obj;
            }, {});

            // Chunk 2: Second half of the keys
            const chunk2Keys = keys.slice(midpoint);
            const chunk2 = chunk2Keys.reduce((obj, key) => {
                obj[key] = baseTranslations[key];
                return obj;
            }, {});

            const chunks = [chunk1, chunk2];
            const results = [];

            // Loop through chunks
            for (const chunk of chunks) {
                try {
                    const prompt = createTranslationPrompt(chunk, targetLangName);
                    let rawText = await generateOutput(prompt);
                    
                    // Clean up response (remove markdown code blocks if present)
                    if (rawText.startsWith('```json')) {
                        rawText = rawText.replace(/```json/g, '').replace(/```/g, '');
                    } else if (rawText.startsWith('```')) {
                        rawText = rawText.replace(/```/g, '');
                    }

                    const chunkTranslation = JSON.parse(rawText);
                    results.push(chunkTranslation);
                } catch (ex) {
                    console.error(`Chunk was not included in translation due to error: ${ex}`);
                    // Fallback: use original english for this chunk if translation fails
                    results.push(chunk);
                }
            }

            // Merge chunks back together
            const finalTranslation = Object.assign({}, ...results);
            
            setTranslations(prev => ({
                ...prev,
                [targetLangCode]: finalTranslation
            }));
            return true;
        } catch (error) {
            console.error("Translation process failed:", error);
            alert("Failed to load translation. Please try again.");
            return false;
        } finally {
            setIsTranslating(false);
        }
    };

    const changeLanguage = async (langCode) => {
        const langConfig = languageOptions.find(l => l.code === langCode);
        if (!langConfig) return;

        // If we don't have it, fetch it
        if (!translations[langCode]) {
            const success = await fetchTranslation(langCode, langConfig.name);
            if (!success) return;
        }

        setLanguage(langCode);
        setDir(langConfig.dir);
        document.documentElement.dir = langConfig.dir;
        document.documentElement.lang = langCode;
    };

    const t = (key) => {
        const currentDict = translations[language] || translations['en'];
        return currentDict[key] || translations['en'][key] || key;
    };

    return (
        <LanguageContext.Provider value={{ language, changeLanguage, dir, t, isTranslating }}>
            {children}
            {isTranslating && (
                <div className="fixed inset-0 z-[200] bg-black/50 backdrop-blur-sm flex items-center justify-center flex-col text-white animate-in fade-in duration-300">
                    <Loader2 className="w-12 h-12 animate-spin mb-4 text-red-500" />
                    <p className="text-xl font-semibold">Translating with Gemini AI...</p>
                </div>
            )}
        </LanguageContext.Provider>
    );
};

const calculateInitialCTAS = (patientData) => {
    let baseCTAS = PatientAcuitData.chief_complaint[patientData.chiefComplaint]?.ctas || 4;
    let adjustment = 0;

    // Severity Adjustment
    if (patientData.severity === 'Severe') {
        adjustment = Math.min(adjustment, -1);
    } else if (patientData.severity === 'Mild') {
        adjustment = Math.max(adjustment, 1);
    }

    // Pain Scale Adjustment (Vital Indicator)
    if (patientData.painLevel >= 8) {
        adjustment = Math.min(adjustment, -1); // High pain increases urgency
    }

    // Automatic Risk Flagging: Age
    const age = parseInt(patientData.age) || 0;
    if (age > 75 || age < 2) {
        adjustment = Math.min(adjustment, -1); // High risk age groups
    }

    // Automatic Risk Flagging: PMH
    const hasHighRiskHistory = Object.keys(PatientAcuitData.pmh_flags).some(
        (key) => patientData.pastMedicalHistory.includes(key)
    );
    if (hasHighRiskHistory && baseCTAS > 2) {
        adjustment = Math.min(adjustment, -1);
    }

    // Critical Chief Complaints hard floor
    let finalCTAS = Math.max(1, Math.min(5, baseCTAS + adjustment));
    if (['Chest Pain', 'Difficulty Breathing'].includes(patientData.chiefComplaint)) {
        finalCTAS = Math.min(2, finalCTAS);
    }
    return finalCTAS;
};

const ERDashboard = ({ db, userId }) => {
    const { t, dir } = useLanguage();
    const [patients, setPatients] = useState([]);
    const [loading, setLoading] = useState(true);
    const [selectedPatient, setSelectedPatient] = useState(null);
    const [isTreatmentMode, setIsTreatmentMode] = useState(false);
    const [filterCtas, setFilterCtas] = useState('All');

    // ... existing helper functions (formatAdmissionTime, calculateWaitTime) ...
    const formatAdmissionTime = (timestamp) => {
        if (!timestamp) return 'N/A';
        const date = timestamp.toDate();
        return date.toLocaleTimeString('en-US', { hour: '2-digit', minute: '2-digit' });
    };

    const calculateWaitTime = (admissionTime) => {
        if (!admissionTime) return '...';
        const now = new Date();
        const admitted = admissionTime.toDate();
        const diffInMs = now - admitted;
        const diffInHours = Math.floor(diffInMs / (1000 * 60 * 60));
        const diffInMinutes = Math.floor((diffInMs % (1000 * 60 * 60)) / (1000 * 60));
        return `${diffInHours}h ${diffInMinutes}m`;
    };

    const handleUpdateStatus = async (docId, newStatus, closeView = true) => {
        try {
            const path = `artifacts/${appId}/public/data/er_patients`;
            await updateDoc(doc(db, path, docId), { status: newStatus });
            if (closeView) {
                setSelectedPatient(null); 
                setIsTreatmentMode(false);
            }
        } catch (error) {
            console.error("Error updating status:", error);
        }
    };

    const handleStartTreatment = async (docId) => {
        // Update status but keep patient selected and switch to Treatment Mode
        await handleUpdateStatus(docId, 'Arrived / In Progress', false);
        setIsTreatmentMode(true);
    };

    const handleDischarge = async (docId) => {
        await handleUpdateStatus(docId, 'Discharged', true);
    };

    const handleClose = () => {
        setSelectedPatient(null);
        setIsTreatmentMode(false);
    };

    useEffect(() => {
        initFirebase();
        authReadyPromise.then(() => {
            const path = `artifacts/${appId}/public/data/er_patients`;
            const q = query(collection(db, path));
            
            const unsubscribe = onSnapshot(q, (snapshot) => {
                const fetchedPatients = snapshot.docs.map(doc => ({
                    id: doc.id,
                    ...doc.data()
                }))
                .sort((a, b) => {
                    // Prioritize active treatment slightly less than CTAS for sorting
                    if (a.ctas !== b.ctas) {
                        return a.ctas - b.ctas; 
                    }
                    if (a.admissionTime && b.admissionTime) {
                         return a.admissionTime.toDate() - b.admissionTime.toDate(); 
                    }
                    return 0;
                });
                
                setPatients(fetchedPatients);
                setLoading(false);
            }, (error) => {
                console.error("Error fetching patients:", error);
                setLoading(false);
            });

            return () => unsubscribe();
        });
    }, [db, userId]);

    // ... existing filteredPatients logic ...
    const filteredPatients = useMemo(() => {
        if (filterCtas === 'All') return patients;
        return patients.filter(p => p.ctas === parseInt(filterCtas));
    }, [patients, filterCtas]);

    if (loading) {
        return (
            <div className="p-8 text-center text-xl font-semibold text-gray-500 animate-pulse">
                Loading...
            </div>
        );
    }

    if (selectedPatient) {
        return (
            <div className="p-6 bg-gray-100 h-full overflow-y-auto">
                <button onClick={handleClose} className="mb-4 text-blue-600 hover:text-blue-800 font-medium flex items-center">
                    ← Back to Dashboard
                </button>
                
                {isTreatmentMode ? (
                    <ActiveTreatmentView 
                        patient={selectedPatient}
                        onClose={handleClose}
                        onDischarge={handleDischarge}
                    />
                ) : (
                    <PatientDetails 
                        patient={selectedPatient} 
                        onClose={handleClose} 
                        onStartTreatment={handleStartTreatment}
                        onDischarge={handleDischarge}
                    />
                )}
            </div>
        );
    }

    return (
        <div dir={dir} className="p-6 bg-gray-50 h-full overflow-y-auto">
            {/* ... existing dashboard rendering ... */}
            <div className="flex flex-col md:flex-row justify-between items-center mb-6 border-b-4 border-red-600 pb-4">
                <h2 className="text-3xl font-bold text-gray-800 flex items-center gap-2">
                    {t('staffDashboard')}
                    <span className="text-sm font-normal bg-red-100 text-red-800 px-2 py-1 rounded-full">{filteredPatients.length} {t('active')}</span>
                </h2>
                
                <div className="flex space-x-1 mt-4 md:mt-0 bg-white p-1 rounded-full shadow-sm border">
                    {['All', '1', '2', '3', '4', '5'].map(filter => (
                        <button
                            key={filter}
                            onClick={() => setFilterCtas(filter)}
                            className={`px-4 py-1 text-sm rounded-full font-semibold transition ${
                                filterCtas === filter 
                                ? 'bg-gray-800 text-white shadow-md' 
                                : 'text-gray-600 hover:bg-gray-100'
                            }`}
                        >
                            {filter === 'All' ? 'All' : `CTAS ${filter}`}
                        </button>
                    ))}
                </div>
            </div>

            <div className="space-y-4">
                {filteredPatients.length === 0 ? (
                    <div className="p-10 text-center text-xl text-gray-500 bg-white rounded-lg shadow-inner border border-gray-200">
                        No patients found.
                    </div>
                ) : (
                    <div className="grid grid-cols-1 gap-4">
                        {filteredPatients.map((patient) => (
                            <div key={patient.id} className="bg-white p-4 rounded-lg shadow-lg border-l-8 border-red-600 hover:shadow-xl transition duration-200">
                                <div className="flex justify-between items-start mb-2">
                                    <h3 className="text-xl font-bold text-gray-900 flex items-center gap-2">
                                        {patient.name} <span className="text-gray-400 font-normal text-sm">({patient.ohipNumber})</span>
                                    </h3>
                                    <TriageTag ctas={patient.ctas} />
                                </div>
                                
                                <div className="grid grid-cols-2 md:grid-cols-4 text-sm text-gray-600 mb-3 border-t pt-2 gap-2">
                                    <div><span className="font-semibold block text-gray-400 text-xs">Time</span> {formatAdmissionTime(patient.admissionTime)}</div>
                                    <div><span className="font-semibold block text-gray-400 text-xs">{t('waitTime')}</span> <span className="text-red-700 font-bold">{calculateWaitTime(patient.admissionTime)}</span></div>
                                    <div><span className="font-semibold block text-gray-400 text-xs">{t('severity')}</span> <span className={`font-bold ${patient.severity === 'Severe' ? 'text-red-600' : 'text-green-600'}`}>{t(patient.severity)}</span></div>
                                    <div><span className="font-semibold block text-gray-400 text-xs">Status</span> {patient.status}</div>
                                </div>

                                <div className="space-y-1 text-sm bg-gray-50 p-2 rounded border border-gray-100">
                                    <p><span className="font-semibold text-gray-500">{t('chiefComplaint')}:</span> <span className="text-red-600 font-medium">{t(patient.chiefComplaint)}</span></p>
                                    <p className="truncate"><span className="font-semibold text-gray-500">Summary:</span> {patient.medicalSummary}</p>
                                </div>

                                <div className="mt-4 flex justify-between items-center border-t pt-3">
                                    <p className="text-xs text-gray-400 font-mono">{patient.id.substring(0,8)}...</p>
                                    
                                    <button 
                                        onClick={() => setSelectedPatient(patient)}
                                        className="px-4 py-2 bg-white hover:bg-blue-50 text-blue-700 text-sm font-bold rounded border border-blue-200 shadow-sm transition flex items-center"
                                    >
                                        {t('viewDetails')} →
                                    </button>
                                </div>
                            </div>
                        ))}
                    </div>
                )}
            </div>
        </div>
    );
};

// 3. Main App Container
const AppContent = () => {
    const { t, language, dir } = useLanguage();
    // State for Login Handling
    const [isLoggedIn, setIsLoggedIn] = useState(false);
    const [patientName, setPatientName] = useState('');
    const [userRole, setUserRole] = useState(null); // Added state for user role

    // 'patient', 'staff', 'hospitals'
    const [currentView, setCurrentView] = useState('patient');
    const [patientSubmitted, setPatientSubmitted] = useState(false);
    const [userId, setUserId] = useState(null);
    const [showLanguagePicker, setShowLanguagePicker] = useState(false);

    useEffect(() => {
        initFirebase();
        authReadyPromise.then(user => {
            setUserId(user?.uid || crypto.randomUUID()); 
        }).catch(err => {
            console.error("Auth failed:", err);
            setUserId(crypto.randomUUID());
        });
    }, []);

    const handleSubmission = () => {
        setPatientSubmitted(true);
        setTimeout(() => {
             setCurrentView('patient'); // Stay on patient view or reset
             setPatientSubmitted(false);
        }, 9000); 
    };

    const handleLogin = (role, name = '') => {
        setIsLoggedIn(true);
        setUserRole(role); // Set role on login
        setCurrentView(role === 'staff' ? 'staff' : 'patient');
        if (name) setPatientName(name);
    };

    const currentLangConfig = languageOptions.find(l => l.code === language);

    // RENDER LOGIN PAGE IF NOT LOGGED IN
    if (!isLoggedIn) {
        return <LandingPage onLogin={handleLogin} />;
    }

    const renderContent = () => {
        if (!userId || !db) {
            return (
                <div className="flex items-center justify-center h-full">
                    <div className="animate-spin rounded-full h-12 w-12 border-b-2 border-red-800"></div>
                    <span className="mx-3 text-gray-600">Loading secure environment...</span>
                </div>
            );
        }

        switch (currentView) {
            case 'patient':
                return <PatientIntake db={db} userId={userId} onSubmission={handleSubmission} initialName={patientName} />;
            case 'staff':
                return <ERDashboard db={db} userId={userId} />;
            case 'hospitals':
                return <HospitalWaitTimes />;
            default:
                return <PatientIntake db={db} userId={userId} onSubmission={handleSubmission} initialName={patientName} />;
        }
    };

    return (
        <div dir={dir} className="min-h-screen bg-gray-100 font-sans antialiased flex flex-col">
            <header className="p-4 bg-red-800 text-white shadow-lg flex flex-col xl:flex-row justify-between items-center sticky top-0 z-50 gap-4">
                <div className="flex items-center gap-2 mb-2 md:mb-0">
                    <span className="text-2xl">🏥</span>
                    <h1 className="text-2xl font-extrabold tracking-tight">{t('appTitle')}</h1>
                </div>
                
                <div className="flex flex-col md:flex-row items-center gap-4">
                    {/* Language Trigger Button */}
                    <button 
                        onClick={() => setShowLanguagePicker(true)}
                        className="flex items-center space-x-2 bg-red-900 hover:bg-red-950 text-white px-4 py-2 rounded-lg transition-colors border border-red-700 shadow-sm"
                    >
                        <Globe className="w-4 h-4" />
                        <span className="font-semibold">{currentLangConfig?.native || 'Language'}</span>
                        <ChevronDown className="w-4 h-4 opacity-70" />
                    </button>

                    {/* View Toggles - Conditionally Rendered based on User Role */}
                    <div className="flex items-center space-x-2 bg-red-900 p-1 rounded-lg">
                        {userRole === 'patient' && (
                            <>
                                <button
                                    onClick={() => setCurrentView('hospitals')}
                                    className={`px-4 py-1.5 rounded-md text-sm font-semibold transition duration-150 ${currentView === 'hospitals' ? 'bg-white text-red-900 shadow-sm' : 'text-red-100 hover:bg-red-800'}`}
                                >
                                    {t('hospitalTimes')}
                                </button>
                                <button
                                    onClick={() => setCurrentView('patient')}
                                    className={`px-4 py-1.5 rounded-md text-sm font-semibold transition duration-150 ${currentView === 'patient' ? 'bg-white text-red-900 shadow-sm' : 'text-red-100 hover:bg-red-800'}`}
                                >
                                    {t('patientView')}
                                </button>
                            </>
                        )}
                        
                        {userRole === 'staff' && (
                            <button
                                onClick={() => setCurrentView('staff')}
                                className={`px-4 py-1.5 rounded-md text-sm font-semibold transition duration-150 ${currentView === 'staff' ? 'bg-white text-red-900 shadow-sm' : 'text-red-100 hover:bg-red-800'}`}
                            >
                                {t('staffDashboard')}
                            </button>
                        )}
                    </div>
                    
                    {/* Logout / Switch User Button */}
                    <button 
                        onClick={() => {
                            setIsLoggedIn(false);
                            setUserRole(null);
                            setPatientName('');
                        }}
                        className="text-xs text-red-200 hover:text-white underline ml-2"
                    >
                        Logout
                    </button>
                </div>
            </header>

            <main className="flex-grow p-4 md:p-6 overflow-hidden flex flex-col">
                {renderContent()}
            </main>

            {/* Language Modal */}
            {showLanguagePicker && (
                <LanguagePicker onClose={() => setShowLanguagePicker(false)} />
            )}

            <footer className="p-3 bg-gray-200 text-gray-500 text-xs text-center border-t border-gray-300">
                App ID: {appId} | Session: {userId ? userId.substring(0,8) + '...' : 'Init...'}
            </footer>
        </div>
    );
};

const App = () => (
    <LanguageProvider>
        <AppContent />
    </LanguageProvider>
);

export default App;
